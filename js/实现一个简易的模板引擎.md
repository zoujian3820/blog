## 实现一个简易的模板引擎

```js
String.prototype.handleTemplateWithData = function (context) {
  // return this.replace(/%%(.*?)%%/g, function (match, key) {
  return this.replace(/{{(.*?)}}/g, function (match, key) {
    //	key代表第1个括号匹配的字符串
    /*console.log(match);*/
    return context[key.trim()];
  });
};

const template =
  "<div>" +
  "<span>{{abc}}</span>" +
  '<li tid="{{ids}}">{{hac}}</li>' +
  "</div>";

template.handleTemplateWithData({ abc: "gggg", ids: 9999, hac: "小风" });
```
