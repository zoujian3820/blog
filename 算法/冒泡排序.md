## 冒泡排序
```js
const arr = [5, 3, 4, 1, 2]

// 轮数
for(let i = 0; i < arr.length - 1; i++) {
  // 次数
  for(let y = 0; y < arr.length - 1 - i; y++) {
    // 判断前一个大于后一个数时 进行交换
    if(arr[y] > arr[y + 1]) {
      // 借助第三方变量 交换两个变量的值
      const temp = arr [y]
      arr[y] = arr[y + 1]
      arr[y + 1 ] = temp
    }
  }
}
```
